password = input("Enter Password: ")
u = l = d = s = 0

for ch in password:
    if ch.isupper():
        u = 1
    elif ch.islower():
        l = 1
    elif ch.isdigit():
        d = 1
    else:
        s = 1
i = 0
while i < len(password):
    i += 1

print("Checking Password :", password)
print("First 3 chars :", password[:3])
print("Last 3 chars :", password[-3:])
if u and l and d and s and i >= 8:
    print("Status : Strong Password")
else:
    print("Status : Weak Password")
